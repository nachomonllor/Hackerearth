int Particion(int vector[], int lim_izq, int lim_der)
{
	int izq=lim_izq, der=lim_der, elemPivote=vector[lim_izq];

	while(izq < der)
	{
        while(vector[izq] >= elemPivote && izq < der) izq++;
	    while(vector[der] < elemPivote) der--;

	    if(izq < der)
        {
	        int temp = vector[izq];
        	vector[izq] = vector[der];
	        vector[der] = temp;
        }
	}
	
	vector[lim_izq] = vector[der];
	vector[der] = elemPivote;
	return der;
}
	
	
void quick(int a[], int left, int right)
{
         
    if (left < right)
    {
        int i = Particion(a, left, right);
        quick(a, left, i - 1);
        quick(a, i + 1, right);
    }
}

bool promNight(int boys[], int M , int  girls[], int N)
{
            quick(boys, 0, M - 1);
            quick(girls, 0, N - 1);

			
			//mostrar(boys, M);
			//mostrar(girls, N);

            if ( N < M){   //girls.size() < boys.size())  {
               return false;
           	}

            if (M == N)
            {
                for (int k = 0; k < M; k++)
                {
                    if (boys[k] < girls[k])
                    {
                        return false;
                    }
                }
            }

            int i = 0, iGirls = 0;
            while (i < M)
            {
                while (iGirls < N && boys[i] <= girls[iGirls])
                {
                    iGirls++;
                }
                
                if (i < M && iGirls > N)
                {
                    return false;
                }
				i++;
            }
            return true;
}
